 <Items>
                        <div className='items-grid'>
                           <span>Item List</span>
                           <span>Qty.</span>
                           <span>Price</span>
                           <span>Total</span>
                        </div>

                        {items.map((item) => {
                           return (
                              <div className='items-grid' key={item.id}>
                                 <input
                                    aria-label='item.name'
                                    name='name'
                                    type='text'
                                    value={item.name}
                                    onChange={(e) =>
                                       setItems((prev) => {
                                          let tempItems = prev.map(
                                             (prevItem) => {
                                                if (prevItem.id === item.id) {
                                                   return {
                                                      ...prevItem,
                                                      name: e.target.value,
                                                   };
                                                } else {
                                                   return prevItem;
                                                }
                                             }
                                          );
                                          return tempItems;
                                       })
                                    }
                                 />
                                 <input
                                    aria-label='quantity'
                                    name='quantity'
                                    type='number'
                                 />
                                 <input
                                    aria-label='price'
                                    name='item.price'
                                    type='number'
                                 />
                                 <div>{item.total.toFixed(2)}</div>

                                 <button
                                    aria-label='delete btn'
                                    type='button'
                                    name='delete btn'
                                    className='delete-btn'
                                    onClick={() => deleteItem(item.id)}
                                 >
                                    <svg
                                       width='13'
                                       height='16'
                                       xmlns='http://www.w3.org/2000/svg'
                                    >
                                       <path
                                          d='M11.583 3.556v10.666c0 .982-.795 1.778-1.777 1.778H2.694a1.777 1.777 0 01-1.777-1.778V3.556h10.666zM8.473 0l.888.889h3.111v1.778H.028V.889h3.11L4.029 0h4.444z'
                                          fill='#888EB0'
                                          fillRule='nonzero'
                                       />
                                    </svg>
                                 </button>
                              </div>
                           );
                        })}

                        <div
                           onClick={() =>
                              setItems((prev) => [
                                 ...prev,
                                 {
                                    id: new Date().getTime().toString(),
                                    name: '',
                                    quantity: '',
                                    price: '',
                                    total: 0,
                                 },
                              ])
                           }
                           className='add-btn'
                        >
                           <svg
                              width='11'
                              height='11'
                              xmlns='http://www.w3.org/2000/svg'
                           >
                              <path
                                 d='M6.313 10.023v-3.71h3.71v-2.58h-3.71V.023h-2.58v3.71H.023v2.58h3.71v3.71z'
                                 fill='#7C5DFA'
                                 fillRule='nonzero'
                              />
                           </svg>
                           Add New Item{' '}
                        </div>
                     </Items>
